 * CommentController.js
 *
 * @description ::
 * @docs        :: http://sailsjs.org/#!documentation/controllers
 */
 
module.exports = {
  create: function (req, res, next) {
    'use strict';
    var badguy = false;
 
    // block bots
    if(req.param('address') !== 'http://') {
      badguy = true;
      return res.badRequest();
    }
 
    var tarpFields = ['contact', 'message'];
 
    _.each(tarpFields, function (tarp) {
      if(req.param(tarp).length > 0 && req.param(tarp).trim().length > 0){
        badguy = true;
        return res.badRequest();
      }
    });
 
    if(!badguy){
      Comment.create(req.params.all())
      .exec(function (err, comment) {
        if (err){
          return next(err);
        }
        console.log("Comment.person: ", comment.person);
        Person.findOne({id: comment.person}).exec(function (err, person) {
          if (err){
            return next(err);
          }
          var s = '/person/'+person.cleanName;
          res.redirect(s);
        });
      });
    }
  },
  destroy: function(req, res) {
    Comment.findOne(req.param('id'))
    .populate('person')
    .exec(function (err, comment) {
      if(err){
        return next(err);
      }
      var personId = comment.person.id;
      var personName = comment.person.cleanName;
      comment.destroy(function (err) {
        if(err){
          return next(err);
        }
        res.redirect('/person/'+ personName);
      });
    });
  }
};
