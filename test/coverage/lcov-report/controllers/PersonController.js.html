/**

 *
 * @description :: Server-side logic for managing people
 * @help        :: See http://sailsjs.org/#!/documentation/concepts/Controllers
 */
 
module.exports = {
  search: function(req, res) {
    'use strict';
 
    Person.count().exec(function(err, found){
      if (err) {
        return res.negotiate(err);
      }
      if (!found) {
        return res.notFound();
      }
      Person.find({
        or : [
          {
            name: {
              'contains': req.param('searchCriteria')
            },
          },
          {
            title: {
              'contains': req.param('searchCriteria')
            }
          }
        ],
        limit: 10
      })
      .populate('companies')
      .populate('projects')
      .populate('comments')
      .exec(function(err, people){
        if (req.wantsJSON) {
          return res.json({
            options: {
              totalPeople: found,
              updatedPeople: people
            }
          });
        } else {
          res.view({
            people: people,
            personTitle: 'Person Administration'});
        }
      });
    });
  },
  show: function(req, res, next){
    'use strict';
    Person.findOne({cleanName: req.param('cleanName') })
      .populate('projects')
      .populate('comments')
      .populate('companies')
      .then(function (person) {
        return [person, person.comments, person.projects, person.companies];
      })
      .spread(function (person, comments) {
        res.view({person: person, comments: comments, personTitle: person.title});
      })
      .fail(function(err){
        if (err) {
          return next(err);
        }
        res.redirect('/');
      });
  },
  index: function(req, res, next) {
    'use strict';
    Person.find()
    .populate('companies')
      .exec(function (err, people) {
      if (err) {
        return next(err);
      }
 
      res.view({people: people, personTitle: 'Person Administration'});
    });
  },
  getPeople: function(req, res, next) {
    'use strict';
    Person.find().exec(function (err, people) {
      if (err) {
        return next(err);
      }
 
      res.json({people: people});
    });
  },
  'new': function (req, res, next) {
    'use strict';
    var person = { 'name': '', 'title': '', cleanName: ''};
    Company.find()
    .exec(function(err, companies){
      Project.find()
      .exec(function(err, projects){
        if (err) {
          return next(err);
        }
        res.view({
          personTitle: 'New Person',
          person: person,
          projects: projects,
          companies: companies,
          error: err
        });
      });
    });
  },
  create: function (req, res, next) {
    'use strict';
    Person.create(req.params.all(), function (err, person) {
      if (err) {
        return next(err);
      }
      Company.find()
      .populate('people')
      .exec(function(err, allCompanies) {
        Project.find()
        .populate('people')
        .exec(function(err, allProjects){
          Person.find()
          if (err) {
            return next(err);
          }
 
          var addList = req.param('addProjects') || [];
          var addCompanyList = req.param('addCompanies') || [];
 
          for (let value of allProjects) {
            if (addList.indexOf(value.id.toString()) === -1){
              person.projects.remove(value.id);
            }
          }
 
          for (let value of allCompanies) {
            if (addCompanyList.indexOf(value.id.toString()) === -1){
              person.companies.remove(value.id);
            }
          }
 
          person.projects.add(req.param('addProjects'));
          person.companies.add(req.param('addCompanies'));
 
          person.save(function(err, updatedPerson){
 
            if (err) {
              // TODO: This is throwing and error that I don't understand. Additionally, association changes don't appear until refresh.
              console.log("Error: ", err);
            }
            return res.redirect('person/'+ person.cleanName);
          });
        });
      });
    });
  },
  edit: function(req, res, next) {
    'use strict';
    Person.findOne({cleanName: req.param('cleanName') })
    .populate('projects')
    .exec(function(err, person){
      Company.find()
      .populate('people')
      .exec(function(err, companies) {
        Project.find()
        .populate('people')
        .exec(function(err, projects) {
          if (err){
            return next(err);
          }
          person = person || {};
          console.log("person.projects in edit: ", person.projects);
          console.log("person.companies in edit: ", person.companies);
          res.view({
            personTitle: 'Edit Person - '+person.title,
            person: person,
            projects: projects,
            companies: companies,
            error: err
          });
        });
      });
    });
  },
  update: function(req, res, next) {
    'use strict';
 
    Person.findOne({cleanName: req.param('cleanName')}, function (err, person){
      if (err) {
        return next(err);
      }
      Company.find()
      .populate('people')
      .exec(function(err, allCompanies) {
        Project.find()
        .populate('people')
        .exec(function(err, allProjects){
          Person.find()
          if (err) {
            return next(err);
          }
 
          var addList = req.param('addProjects') || [];
          var addCompanyList = req.param('addCompanies') || [];
 
          for (let value of allProjects) {
            if (addList.indexOf(value.id.toString()) === -1){
              person.projects.remove(value.id);
            }
          }
 
          for (let value of allCompanies) {
            if (addCompanyList.indexOf(value.id.toString()) === -1){
              person.companies.remove(value.id);
            }
          }
 
          person.projects.add(req.param('addProjects'));
          person.companies.add(req.param('addCompanies'));
 
          person.save(function(err, savedPerson){
 
            if (err) {
              // TODO: This is throwing and error that I don't understand. Additionally, association changes don't appear until refresh.
              console.log("Error: ", err);
            }
            Person.update({cleanName: req.param('cleanName')}, req.params.all(), function (err, people){
              res.redirect('/person/' + people[0].cleanName);
            });
          });
        });
      });
    });
  },
  destroy: function(req, res, next) {
    'use strict';
    Person.findOne({cleanName: req.param('cleanName') }).exec(function(err, person){
      if (err){
        return next(err);
      }
      person = person || {};
 
      person.destroy(function (err) {
        res.redirect('/person');
      });
    });
  }
};
